41  /**
42   * <p>A Yaml based implementation of the {@link ConfigurationParser}.</p>
43   * <p>The yaml configuration file need to have the root keys be the same 
44   * as the environment name.  The next level key needs to match the name 
45   * of a key that the configuration uses in the {@link CadmiumConfig} annotation. 
46   * All configuration elements below that level need to match the field names 
47   * and types or the pojo class that was wired in.</p>
48   * <p>The following is an example of a yaml configuration file:<br />
49   * <pre>
50   * default:
51   *   email: !email
52   *     &email
53   *     jndiName: 'java:/Mail'
54   *     sessionStrategy: 'com.meltmedia.cadmium.email.JndiSessionStrategy'
55   *     messageTransformer: 'com.meltmedia.cadmium.email.IdentityMessageTransformer'
56   * production:
57   *   email: !email
58   *     <<: *email
59   *     jndiName: 'java/ProdMail'
60   * </pre>
61   * </p>
62   * @author John McEntire
63   *
64   */
65  public class YamlConfigurationParser implements ConfigurationParser {

68     * The environment key to be used for all default values.

70    public static final String DEFAULT = "default";

74    @SuppressWarnings("rawtypes")
75    protected List<Class> configurationClasses = new ArrayList<Class>();
76    protected Map<String, Map<String, ?>> configuration = new HashMap<String, Map<String, ?>>();
77    protected String environment;

81    /**
82     * This will only parse the files with the extensions of <code>.yml</code> or <code>.yaml</code> in the directory specified.
83     */
84    @Override
85    public void parseDirectory(File configurationDirectory) throws Exception {
86      if(configurationDirectory != null && configurationDirectory.isDirectory() && configurationDirectory.canRead()) {
87        Collection<File> configFiles = FileUtils.listFiles(configurationDirectory, new String[] {"yml", "yaml"}, true);
88        List<File> files = new ArrayList<File>(configFiles);
89        Collections.sort(files, NameFileComparator.NAME_INSENSITIVE_COMPARATOR);
90        Map<String, Map<String, ?>> configurationMap = new HashMap<String, Map<String, ?>>();
91        Yaml yamlParser = new Yaml(getClassTags());
92        for(File configFile : files) {
93          FileReader reader = null;
94          try {
95            reader = new FileReader(configFile);
96            for(Object parsed : yamlParser.loadAll(reader)){
97              mergeConfigs(configurationMap, parsed);
98            }
99          } finally {
100           IOUtils.closeQuietly(reader);
101         }
102       }
103       this.configuration = configurationMap;
104     } else if(configurationDirectory != null && configurationDirectory.isDirectory()){
105       logger.warn("Directory {} cannot be read.", configurationDirectory);
106     } else if(configurationDirectory != null){
107       logger.warn("Directory {} is not a directory or does not exist.", configurationDirectory);
108     } else {
109       throw new IllegalArgumentException("The configurationDirectory must be specified.");
110     }
111   }

113   /**
114    * Merges configurations from an Object into an existing Map.
115    * 
116    * @param configurationMap
117    * @param parsed 
118    */
119   @SuppressWarnings("unchecked")
120   private void mergeConfigs(Map<String, Map<String, ?>> configurationMap,
121       Object parsed) {
122     if(parsed instanceof Map) {
123       Map<?, ?> parsedMap = (Map<?, ?>) parsed;
124       for(Object key : parsedMap.keySet()) {
125         if(key instanceof String) {
126           Map<String, Object> existingValue = (Map<String, Object>) configurationMap.get((String) key);
127           if(!configurationMap.containsKey((String) key)) {
128             existingValue = new HashMap<String, Object>();
129             configurationMap.put((String) key, existingValue);
130           }
131           Object parsedValue = parsedMap.get(key);
132           if(parsedValue instanceof Map) {
133             Map<?, ?> parsedValueMap = (Map<?, ?>) parsedValue;
134             for(Object parsedKey : parsedValueMap.keySet()) {
135               if(parsedKey instanceof String) {
136                 existingValue.put((String) parsedKey, parsedValueMap.get(parsedKey));
137               }
138             }
139           }
140         }
141       }
142     }
143   }

145   /**
146    * 
147    * @return A new Representer Instance with the tags specified by the {@link configurationClasses} list.
148    */
149   private Constructor getClassTags() {
150     Constructor constructor = new YamlLenientConstructor();
151     if(configurationClasses != null) {
152       for(Class<?> configClass : configurationClasses) {
153         CadmiumConfig configAnnotation = configClass.getAnnotation(CadmiumConfig.class);
154         if(configAnnotation != null) {
155           String key = configAnnotation.value();
156           if(StringUtils.isEmptyOrNull(key)) {
157             key = configClass.getCanonicalName();
158           }
159           
160           if(key != null) {
161             constructor.addTypeDescription(new TypeDescription(configClass, "!" + key));
162             logger.debug("Adding configuration tag {} for class {}", "!"+key, configClass);
163           }
164         }
165       }
166     }
167     return constructor;
168   }

170   @Override
171   public <T> T getConfiguration(String key, Class<T> type)
172       throws ConfigurationNotFoundException {
173     Map<String, ?> envMap = configuration.get(environment);
174     Map<String, ?> defaultMap = configuration.get(DEFAULT);
175     if(envMap != null && envMap.containsKey(key)) {
176       T config = getEnvConfig(envMap, key, type);
177       if(config != null) {
178         return config;
179       }
180     }
181     if(defaultMap != null && defaultMap.containsKey(key)) {
182       T config = getEnvConfig(defaultMap, key, type);
183       if(config != null) {
184         return config;
185       }
186     }
187     
188     throw new ConfigurationNotFoundException("No configuration with the key \""+key+"\" and type \""+type+"\" were found in environment \""+environment+"\".");
189   }

191   /**
192    * Helper method used to retrieve the value from a given map if it matches the type specified.
193    * 
194    * @param configs The map to pull from.
195    * @param key The key to pull.
196    * @param type The expected type of the value.
197    * @return
198    */
199   private <T> T getEnvConfig(Map<String, ?> configs, String key, Class<T> type) {
200     if(configs.containsKey(key)) {
201       Object config = configs.get(key);
202       if(type.isAssignableFrom(config.getClass())) {
203         return type.cast(config);
204       }
205     }
206     return null;
207   }
