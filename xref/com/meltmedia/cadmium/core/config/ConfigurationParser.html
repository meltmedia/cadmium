16  package com.meltmedia.cadmium.core.config;
17  
18  import java.io.File;
19  import java.util.Collection;
20  
21  /**
22   * <p>An interface of a Configuration Parser that will be created for each configuration update. 
23   * An implementation of this interface must have a zero argument constructor. This will be run 
24   * in the following steps:<br />
25   *   <ol>
26   *     <li>The implementation of this interface will be initialized.</li>
27   *     <li>The {@link ConfigurationParser#setConfigurationClasses(Collection)} will be invoked 
28   *     with a List of Class Objects. (<em>The Class Objects passed in will be wired via Guice 
29   *     and must be Pojo's annotated with {@link CadmiumConfig}.</em>)</li>
30   *     <li>The {@link ConfigurationParser#setEnvironment(String)} will be invoked with the token
31   *      for the current environment.</li>
32   *     <li>The {@link ConfigurationParser#parseDirectory(File)} method will be called with a File
33   *      Object that will point to the directory that holds the configuration files.</li>
34   *     <li>When ever a module needs access to a configuration the call will be delegated to the 
35   *     {@link ConfigurationParser#getConfiguration(String, Class)} method.</li> 
36   *   </ol>
37   * </p>
38   * 
39   * @author John McEntire
40   *
41   */
42  public interface ConfigurationParser {
43    
44    /**
45     * Sets the classes that will be expected in the configuration files.
46     * 
47     * @param configurationClasses
48     */
49    @SuppressWarnings("rawtypes")
50    public void setConfigurationClasses(Collection<Class> configurationClasses);
51    
52    /**
53     * Sets the environment token for this instances environment.
54     * 
55     * @param environment
56     */
57    public void setEnvironment(String environment);
58    
59    /**
60     * This method will process all configuration files that are contained in the directory specified.
61     * 
62     * @param configurationDirectory A File Object that references the directory where all 
63     * configuration file will be stored.
64     * @throws Exception 
65     */
66    public void parseDirectory(File configurationDirectory) throws Exception;
67    
68    /**
69     * This method will be used to fetch a configuration from the freshly parsed configurations.
70     * 
71     * @param key The key that references the configuration wanted.
72     * @param type The type of the Pojo that the configuration was parsed in as.
73     * @return The instance of the Pojo that is created off of the configuration parsed.
74     * @throws ConfigurationNotFoundException Thrown if the configuration requested does not exist.
75     */
76    public <T> T getConfiguration(String key, Class<T> type) throws ConfigurationNotFoundException;
77  }
